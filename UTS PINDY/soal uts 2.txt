#include <iostream>
using namespace std;

int main() {
    // a) Inisialisasi tabungan dalam Dollar dan Euro
    double tabungan_dollar = 12000.0;
    double tabungan_euro = 20000.0;

    // b) Kurs mata uang
    double kurs_usd_to_idr = 15000.0;
    double kurs_euro_to_usd = 0.997;

    // Konversi tabungan ke Rupiah
    double total_tabungan_idr = (tabungan_dollar * kurs_usd_to_idr) + ((tabungan_euro * kurs_euro_to_usd) * kurs_usd_to_idr);

    // c) Harga mobil dan perhitungan sisa uang
    double harga_mobil = 425000000.0;
    double sisa_uang = total_tabungan_idr - harga_mobil;

    // Menampilkan hasil perhitungan
    cout << "Total tabungan dalam Rupiah: " << total_tabungan_idr << endl;
    cout << "Sisa uang setelah membeli mobil: " << sisa_uang << endl;

    return 0;
}
